\begin{align*}
    \ket{\psi_0} &= \ket{xy} \\ \\ \\
    \ket{\psi_1} &= H_1\ket{\psi_0} \\
    \ket{\psi_1} &= \frac{1}{\sqrt{2}}(\ket{0} + (-1)^x\ket{1})\ket{y} \\
    \ket{\psi_1} &= \frac{1}{\sqrt{2}}(\ket{0y} + (-1)^x\ket{1y}) \\ \\ \\
    \ket{\psi_2} &= CNOT_{1\rightarrow2}\ket{\psi_1} \\
    \ket{\psi_2} &= \frac{1}{\sqrt{2}}(CNOT_{1\rightarrow2}\ket{0y} + (-1)^x CNOT_{1\rightarrow2}\ket{1y}) \\
    \ket{\psi_2} &= \frac{1}{\sqrt{2}}(\ket{0 \quad 0 \oplus y} + (-1)^x\ket{1 \quad 1 \oplus y}) \\
    \ket{\psi_2} &= \frac{1}{\sqrt{2}}(\ket{0y} + (-1)^x\ket{1 \overline{y}}) \\ \\ \\
    \ket{\psi_3} &= CNOT_{1\rightarrow2}(\ket{\psi_2}) \\
    \ket{\psi_3} &= \frac{1}{\sqrt{2}}(CNOT_{1\rightarrow2}\ket{0y} + (-1)^x CNOT_{1\rightarrow2}\ket{1\overline{y}}) \\
    \ket{\psi_3} &= \frac{1}{\sqrt{2}}(\ket{0 \quad 0 \oplus y} + (-1)^x \ket{1 \quad 1 \oplus \overline{y}}) \\ 
    \ket{\psi_3} &= \frac{1}{\sqrt{2}}(\ket{0y} + (-1)^x \ket{1y}) \\ \\ \\
    \ket{\psi_4} &= H_1\ket{\psi_3} \\
    \ket{\psi_4} &= \frac{1}{\sqrt{2}}(H\ket{0}\ket{y} + (-1)^xH\ket{1}\ket{y}) \\
    \ket{\psi_4} &= \frac{1}{2}((\ket{0} + \ket{1})\ket{y} + (-1)^x(\ket{0} - \ket{1})\ket{y}) \\
    \ket{\psi_4} &= \frac{1}{2}(\ket{0y} +\ket{1y} + (-1)^x(\ket{0y} - \ket{1y})) \\
\end{align*}

Évaluons pour $\ket{\psi} = \ket{00}$: \\
\begin{align*}
    \ket{\psi} &= \frac{1}{2}(\ket{00} + \ket{10} + \ket{00} - \ket{10}) \\
    \ket{\psi} &= \frac{1}{2}(2\cdot\ket{00}) \\
    \ket{\psi} &= \ket{00}
\end{align*}

Évaluons pour $\ket{\psi} = \ket{01}$: \\
\begin{align*}
    \ket{\psi} &= \frac{1}{2}(\ket{01} + \ket{10} + \ket{01} - \ket{10}) \\
    \ket{\psi} &= \frac{1}{2}(2\cdot\ket{01}) \\
    \ket{\psi} &= \ket{01}
\end{align*}

Évaluons pour $\ket{\psi} = \ket{10}$: \\
\begin{align*}
    \ket{\psi} &= \frac{1}{2}(\ket{00} + \ket{10} - \ket{00} + \ket{10}) \\
    \ket{\psi} &= \frac{1}{2}(2\cdot\ket{10}) \\
    \ket{\psi} &= \ket{10}
\end{align*}
